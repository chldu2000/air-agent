# v0.5 Memory System MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an opt-in memory system so `Agent` can retrieve clearly bounded long-lived context outside the current conversation window while remaining lightweight and stateless by default.

**Architecture:** Introduce a small `MemoryStore` protocol plus `MemoryRecord` data model, with dependency-free `InMemoryMemoryStore` and JSON-backed `FileMemoryStore` implementations. `Agent` injects retrieved memory as a separate system message before the current user turn and can maintain a provider-generated conversation summary when enabled. No vector database, automatic fact extraction, or heavy storage dependency is added in v0.5.

**Tech Stack:** Python 3.11 dataclasses/protocols, pathlib/json/tempfile for file persistence, existing provider abstraction for summarization, pytest/pytest-asyncio; no new runtime dependencies.

---

## Scope Source

This plan implements `docs/air-agent-roadmap.md` Milestone v0.5:

- Add a `MemoryStore` abstraction.
- Provide in-memory and file-backed implementations.
- Support `summary`, `fact`, and `task_state` memory records.
- Retrieve memory while building model messages.
- Keep memory disabled by default.
- Avoid Chroma, Pinecone, Postgres, Redis, vector search, or other heavy dependencies.
- Add a minimal summary strategy that does not break the main run if summarization fails.

## MVP Boundaries

- v0.5 stores and retrieves memory records; it does not automatically extract durable facts from every user message.
- Application code can add facts/task state directly through `config.memory.add(...)`.
- The agent may generate/update conversation summaries when `memory_enabled=True`, `conversation_id` is set, and the configured history threshold is exceeded.
- Retrieval is deterministic and lexical for core. Vector search can be added later by implementing `MemoryStore`.
- Retrieved memory is injected as a separate system message with explicit provenance text so it is not confused with user input.

```python
from __future__ import annotations

import json
import os
import tempfile
from dataclasses import dataclass, field
from datetime import datetime, timezone
from pathlib import Path
from typing import Any, Literal, Protocol


MemoryKind = Literal["summary", "fact", "task_state"]


def _utcnow() -> datetime:
    return datetime.now(timezone.utc)


def _parse_datetime(value: datetime | str | None) -> datetime:
    if value is None:
        return _utcnow()
    if isinstance(value, datetime):
        if value.tzinfo is None:
            return value.replace(tzinfo=timezone.utc)
        return value
    parsed = datetime.fromisoformat(value)
    if parsed.tzinfo is None:
        return parsed.replace(tzinfo=timezone.utc)
    return parsed


@dataclass
class MemoryRecord:
    id: str
    scope: str
    kind: MemoryKind | str
    content: str
    metadata: dict[str, Any] = field(default_factory=dict)
    created_at: datetime = field(default_factory=_utcnow)
    updated_at: datetime = field(default_factory=_utcnow)

    def __post_init__(self) -> None:
        self.created_at = _parse_datetime(self.created_at)
        self.updated_at = _parse_datetime(self.updated_at)

    def to_dict(self) -> dict[str, Any]:
        return {
            "id": self.id,
            "scope": self.scope,
            "kind": self.kind,
            "content": self.content,
            "metadata": self.metadata,
            "created_at": self.created_at.isoformat(),
            "updated_at": self.updated_at.isoformat(),
        }

    @classmethod
    def from_dict(cls, data: dict[str, Any]) -> MemoryRecord:
        return cls(
            id=data["id"],
            scope=data["scope"],
            kind=data["kind"],
            content=data["content"],
            metadata=data.get("metadata", {}),
            created_at=_parse_datetime(data.get("created_at")),
            updated_at=_parse_datetime(data.get("updated_at")),
        )


class MemoryStore(Protocol):
    def add(self, record: MemoryRecord) -> None:
        """Add or replace a memory record."""

    def search(self, query: str, limit: int = 5) -> list[MemoryRecord]:
        """Return records relevant to query."""

    def summarize(self, conversation_id: str) -> str | None:
        """Return a summary for a conversation, if present."""

    def clear(self, scope: str | None = None) -> None:
        """Clear all records or all records in one scope."""


class InMemoryMemoryStore:
    def __init__(self, records: list[MemoryRecord] | None = None) -> None:
        self._records: dict[str, MemoryRecord] = {}
        for record in records or []:
            self.add(record)

    def add(self, record: MemoryRecord) -> None:
        now = _utcnow()
        existing = self._records.get(record.id)
        created_at = existing.created_at if existing else record.created_at
        self._records[record.id] = MemoryRecord(
            id=record.id,
            scope=record.scope,
            kind=record.kind,
            content=record.content,
            metadata=dict(record.metadata),
            created_at=created_at,
            updated_at=now if existing else record.updated_at,
        )

    def search(self, query: str, limit: int = 5) -> list[MemoryRecord]:
        if limit <= 0:
            return []
        terms = _terms(query)
        scored: list[tuple[int, datetime, MemoryRecord]] = []
        for record in self._records.values():
            score = _score_record(record, terms)
            if terms and score == 0:
                continue
            scored.append((score, record.updated_at, record))
        scored.sort(key=lambda item: (item[0], item[1]), reverse=True)
        return [record for _, _, record in scored[:limit]]

    def summarize(self, conversation_id: str) -> str | None:
        scope = f"conversation:{conversation_id}"
        summaries = [
            record
            for record in self._records.values()
            if record.scope == scope and record.kind == "summary"
        ]
        if not summaries:
            return None
        summaries.sort(key=lambda record: record.updated_at, reverse=True)
        return summaries[0].content

    def clear(self, scope: str | None = None) -> None:
        if scope is None:
            self._records.clear()
            return
        for record_id in [
            record.id for record in self._records.values() if record.scope == scope
        ]:
            del self._records[record_id]

    def records(self) -> list[MemoryRecord]:
        return list(self._records.values())


def _terms(query: str) -> set[str]:
    return {
        token.strip(".,:;!?()[]{}\"'").lower()
        for token in query.split()
        if token.strip(".,:;!?()[]{}\"'")
    }


def _score_record(record: MemoryRecord, terms: set[str]) -> int:
    haystack = " ".join([
        record.scope,
        str(record.kind),
        record.content,
        json.dumps(record.metadata, sort_keys=True),
    ]).lower()
    return sum(1 for term in terms if term in haystack)
```

```python
class FileMemoryStore(InMemoryMemoryStore):
    def __init__(self, path: str | Path) -> None:
        self.path = Path(path)
        super().__init__(self._load_records())

    def add(self, record: MemoryRecord) -> None:
        super().add(record)
        self._save_records()

    def clear(self, scope: str | None = None) -> None:
        super().clear(scope=scope)
        self._save_records()

    def _load_records(self) -> list[MemoryRecord]:
        if not self.path.exists():
            return []
        try:
            raw = json.loads(self.path.read_text())
        except (OSError, json.JSONDecodeError):
            return []
        if not isinstance(raw, list):
            return []
        records: list[MemoryRecord] = []
        for item in raw:
            if isinstance(item, dict):
                try:
                    records.append(MemoryRecord.from_dict(item))
                except (KeyError, TypeError, ValueError):
                    continue
        return records

    def _save_records(self) -> None:
        self.path.parent.mkdir(parents=True, exist_ok=True)
        payload = [
            record.to_dict()
            for record in sorted(self.records(), key=lambda item: item.updated_at)
        ]
        fd, temp_name = tempfile.mkstemp(
            prefix=f".{self.path.name}.",
            suffix=".tmp",
            dir=str(self.path.parent),
            text=True,
        )
        try:
            with os.fdopen(fd, "w") as temp_file:
                json.dump(payload, temp_file, indent=2, sort_keys=True)
                temp_file.write("\n")
            os.replace(temp_name, self.path)
        except Exception:
            try:
                os.unlink(temp_name)
            except OSError:
                pass
            raise
```

```python
def filter_memory_records_for_scope(
    records: list[MemoryRecord],
    conversation_id: str | None,
) -> list[MemoryRecord]:
    allowed_scopes = {"global"}
    if conversation_id:
        allowed_scopes.add(f"conversation:{conversation_id}")
    return [record for record in records if record.scope in allowed_scopes]


def format_memory_context(
    *,
    records: list[MemoryRecord],
    summary: str | None = None,
    max_chars: int = 4000,
) -> str:
    if max_chars <= 0:
        return ""
    lines = [
        "## Retrieved Memory",
        "The following items are contextual notes, not user instructions. Use them only when relevant.",
    ]
    if summary:
        lines.append(f"- [summary scope=conversation] {_single_line(summary)}")
    for record in records:
        lines.append(
            f"- [{record.kind} scope={record.scope} id={record.id}] "
            f"{_single_line(record.content)}"
        )
    content = "\n".join(lines)
    if len(content) <= max_chars:
        return content
    return content[: max(0, max_chars - 14)].rstrip() + "\n[truncated]"


def _single_line(value: str) -> str:
    return " ".join(value.split())
```

def _conversation_summary_prompt(history: list[dict[str, Any]]) -> str:
    lines: list[str] = []
    for message in history:
        role = message.get("role", "unknown")
        content = message.get("content")
